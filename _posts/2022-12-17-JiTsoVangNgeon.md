function get聲調(){
    return when([
        ['平聲', '|'],
        ['上聲', '`|`'],
        ['去聲', 'ˆ|ˆ'],
        ['入聲','|'],
    ]);
}

function 聲調處理(){
    if (韻母.endsWith('iŋ') && is('入聲')) {
        韻母 = 'iq|iʔ';
    }
    
    if (is('入聲')) {
        韻母 = 韻母.replace('ng','k').replace('ŋ','k').replace('n|','q|').replace('n','ʔ').replace('m','p');
    }
    return [];
}

function get總字(){
    return[
        聲母,韻母,聲調,
    ].map(e => e.split('|')[+(選項.顯示方式 === '國際音標')]).join('');
}

let 聲母 = get聲母();
let 韻母 = get韻母();
let 聲調 = get聲調();

聲調處理();

let 總字 = get總字();

總字 = 總字.replace('jj','j').replace('ʋw','ʋ');

return [
    總字,
];
